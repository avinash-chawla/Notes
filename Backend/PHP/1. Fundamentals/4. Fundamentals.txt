4. Arrays

1. Indexed Arrays
2. Associated Arrays
3. Multi dimensional Arrays

<?php

   $peoples = [ 'shaun', 'crystal', 'ryu' ];

   $peoples[0];

   $peoples2 = array('ken', 'chun-li');    

   $age = [20, 40, 60];

   print_r($age);		--> print readable version of array

   $age[1] = 50;		--> Changing item at index '1'

   $age[] = 60;			--> append item to array

   array_push($age, 70);	--> append item to array

   count($age)			--> length of array

   $peoplemerged = array_merge($peoples, $peoples2)

?>

5. Associative Arrays

<?php

   $ninjasOne = ['shaun' => 'black',

   $ninjasTwo = array('grapes' => 'green', 'pineapple' => 'yellow');

   $ninjasTwo['toad'] = 'pink'			--> append item in array if not found

6. Loops

<?php

   $ninjas = ['shaun', 'ryu', 'yoshi'];

   for($i = 0; $i < count($ninjas); $i++) {
	echo $ninjas[$i] . '<br />' ;
   }

   foreach($ninjas as $ninja){
	echo $ninjas . '<br />'; 
   }

   $products = [
   	['name' => 'title1', 'price' => 10],
   	['name' => 'title2', 'price' => 20],
   	['name' => 'title3', 'price' => 30],
   	['name' => 'title4', 'price' => 40],
   	['name' => 'title5', 'price' => 50],
   	['name' => 'title6', 'price' => 60]
   ];

   foreach($products as $product){
	echo $product['name'] . '-' . $product['price'];
  	echo '<br />';
   }

   $i = 0;

   while($i < count($products)){
  	echo $products[$i]['name'];
	echo '<br />';
	$i++;
   }

?>

8. Booleans and Comparisons

<?php

    echo true		--> 1
    echo false		--> ""

    echo 5 < 10		--> 1
    echo 5 > 10;	--> ""

    echo 4 == 10;	--> ""
    echo 4 != 10 	--> 1

    echo 'shaun' < 'yoshi'	--> 1  (s < y)

    echo 5 == '5';	--> 1
    echo 5 === '5';	--> ""

    echo true == "1";
    echo false == "";


?>

9. Conditional Statements

<?php

   $price = 20;

   if($price < 10) {
	echo 'The condition is met';

	echo 'elseif condition is met';

	echo 'condition not met';
   }

   foreach($products as $product){

12. Variable Scope

<?php

   function myFunc($age){
	echo $age;
   }

   echo $age;				---> Error (bec. this variable has local scope)

-----------------------------------------------------------------------------------------

   $name = 'maria';

   function sayHello(){
	echo "hello $name";		---> Error (bec. this variable has local scope)
   }

   sayHello();

-----------------------------------------------------------------------------------------

   $name = 'maria';

   function sayHello(){
	global $name;			---> Same as 'this' keyword
	echo "hello $name";		
   }

   sayHello();

-----------------------------------------------------------------------------------------

   $name = 'maria';

   function sayHello(&$name){
	$name = 'wario';
	echo "hello $name";		
   }

   sayHello($name);

?>
