2. Super Global Variables

--> They are special array variables
--> They are already populated with data before your code runs
 
<?php

echo $_SERVER['SERVER_NAME'] . '<br />';		--> localhost
echo $_SERVER['REQUEST_METHOD'] . '<br />';		--> GET
echo $_SERVER['SCRIPT_FILENAME'] . '<br />';		--> C:/xampp/htdocs/sandbox.php
echo $_SERVER['PHP_SELF'] . '<br />';			--> /tuts/sandbox.php

?>

--> Use case:

<form action="<?php echo $_SERVER['PHP_SELF']" method="POST">

--> Even if fileName is changed, this action will point to same file.
