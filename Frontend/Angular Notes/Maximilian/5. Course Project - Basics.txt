2. Recipe Model Class

--> To work with data, we use objects.
--> To ensure that every typescript class uses same kind of object, we need class

--> Class also ensures type of object while using it in project.

------------------------------------------------------------------------------------

export class Recipe{
    name: string;
    description: string;
    imagePath: string;

    constructor(name: string, desc: string, imagePath: string)
    {
        this.name = name;
        this.description = desc;
        this.imagePath = imagePath;
    }
}

------------------------------------------------------------------------------------

3. Using that object

export class RecipeListComponent implements OnInit {
  recipes: Recipe[] = [
    new Recipe('A Test Recipe', 'This is simply a test', 'https://upload.wikimedia.org/wikipedia/commons/1/15/Recipe_logo.jpeg'),
    new Recipe('A Test Recipe', 'This is simply a test', 'https://upload.wikimedia.org/wikipedia/commons/1/15/Recipe_logo.jpeg')
  ];
}

------------------------------------------------------------------------------------

<div *ngFor="let recipe of recipes">
          <h4>{{ recipe.name }}</h4>
          <p>{{ recipe.description }}</p>
	  <img
            [src]="recipe.imagePath"
            alt="{{ recipe.name }}">
</div>
